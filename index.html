       const findEvenOdd = (N, Arr) => 
        {
          let sumEven=0;
          let sumOdd=0;
          for(let i=0; i < N; i++){
              if(Arr[i]%2===0){
                 sumEven += Arr[i];
                }
                else{
                  sumOdd += Arr[i];
                }
            }
          return [sumEven , sumOdd];
    
        };

9.Unique Color Shirt

        function Unique_Shirts (arr,N) {
           let freq=[1001];
   
          for(let i=0; i<1001;i++){
              freq[i]=0;
            }
          for(let i=0;i<N;i++){
              freq[arr[i]]++;
            }
           let ans=0;
            for(let i=0; i<1001; i++){
               if(freq[i]===1){
                  ans++;
                }
            }
           return ans;
        }

10.Min and Max

        function arrayMin(arr) {
         let min=arr[0];
         for(let i=1; i < arr.length;i++){
             if(arr[i]<min){
                  min=arr[i];
                }
            }
             return min;
        }

        function arrayMax(arr) {
         let max=arr[0];
          for(let i=1; i<arr.length;
              i++){
             if(arr[i]>max){
                 max=arr[i];
                }
            }
         return max;
        }
